Qu 1. You are given a string of length n, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string?

    let str="abcadeecfb", newStr=""
    let s = new Set(str);
    for(let i of s){
      newStr+=i
    }
    console.log(newStr)
